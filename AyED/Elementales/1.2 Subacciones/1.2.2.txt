ambiente
n:entero
suma:entero
function sumarNaturales(n:entero):entero :
    mientras n <> 0 hacer
        suma:= suma + (n MOD 10)
        n:= n DIV 10
    fin mientras
    sumarNaturales:=suma;
fin function
proceso
    esc('ingrese su numero natural')
    leer(n)
    suma:=0
    suma:=sumarNaturales(n)
    esc('la suma de los digitos de su numero natural es', suma)
fin accion
